<script>
    // Get all the input elements
    const nameInput = document.getElementById('name');
    const emailInput = document.getElementById('email');
    const addressInput = document.getElementById('address');
    const postcodeInput = document.getElementById('postcode');

    // Add event listeners for Enter key on each field
    nameInput.addEventListener('keydown', (event) => {
        if (event.key === 'Enter') {
            event.preventDefault();
            emailInput.focus();
        }
    });

    emailInput.addEventListener('keydown', (event) => {
        if (event.key === 'Enter') {
            event.preventDefault();
            addressInput.focus();
        }
    });

    addressInput.addEventListener('keydown', (event) => {
        if (event.key === 'Enter') {
            event.preventDefault();
            postcodeInput.focus();
        }
    });

    // Fungsi untuk menghapus nilai input jika nilainya sama dengan teks default
    function clearInput(element, defaultValue) {
        if (element.value === defaultValue) {
            element.value = '';
        }
    }

    // Fungsi untuk mengembalikan nilai default jika input kosong
    function restoreInput(element, defaultValue) {
        if (element.value === '') {
            element.value = defaultValue;
        }
    }

    // Tambahkan event listener untuk fokus pada setiap elemen input
    nameInput.addEventListener('focus', () => clearInput(nameInput, 'you name here'));
    nameInput.addEventListener('blur', () => restoreInput(nameInput, 'you name here'));

    emailInput.addEventListener('focus', () => clearInput(emailInput, 'your email here'));
    emailInput.addEventListener('blur', () => restoreInput(emailInput, 'your email here'));

    addressInput.addEventListener('focus', () => clearInput(addressInput, '25 Nowhere Street\nSomewhere\nNot Here\nY2K 5RU'));
    addressInput.addEventListener('blur', () => restoreInput(addressInput, '25 Nowhere Street\nSomewhere\nNot Here\nY2K 5RU'));
    
    postcodeInput.addEventListener('focus', () => clearInput(postcodeInput, 'postal code'));
    postcodeInput.addEventListener('blur', () => restoreInput(postcodeInput, 'postal code'));

</script>
